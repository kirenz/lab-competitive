# Function to generate monthly data
def generate_monthly_data(eshop_name, start_year, num_years=3):
    # Generate a monthly date range for 3 years
    dates = pd.date_range(start=f'{start_year}-01-01', periods=num_years*12, freq='M')
    
    # Initialize a random number generator
    rng = np.random.default_rng(42)

    # Generate revenues, active user base, ratings, social media followers, and response times
    revenues = np.round(np.linspace(10, 50, num=num_years*12) + rng.normal(scale=3, size=num_years*12), 2)
    time_on_site = np.round(np.linspace(1, 10, num=num_years*12) + rng.normal(scale=1, size=num_years*12), 2)
    ratings = np.clip(np.linspace(3.5, 4.5, num=num_years*12) + rng.normal(scale=0.1, size=num_years*12), 1, 5)
    social_followers = np.round(np.linspace(100, 1000, num=num_years*12) + rng.normal(scale=50, size=num_years*12), 2)
    response_times = np.clip(np.linspace(2, 1, num=num_years*12) + rng.normal(scale=0.1, size=num_years*12), 0.5, 2.5)

    # Create a DataFrame
    data = {
        'eshop_name': eshop_name,
        'date': dates,
        'annual_revenue': revenues,
        'time_on_site': time_on_site,
        'average_rating': ratings,
        'social_media_followers': social_followers,
        'average_response_time': response_times
    }
    df = pd.DataFrame(data)
    return df

# Generate data for E-ShopA, E-ShopB, and E-ShopC
df_a = generate_monthly_data('E-ShopA', start_year=2020)
df_b = generate_monthly_data('E-ShopB', start_year=2020)
df_c = generate_monthly_data('E-ShopC', start_year=2020)

# Combine the data into one DataFrame
df = pd.concat([df_a, df_b, df_c])

# Make numbers different
LIST = ['annual_revenue', 'time_on_site', 'average_rating', 'social_media_followers', 'average_response_time']
LEN = len(df)
DATA = np.random.uniform(0.5, 1.5, size=LEN)

for i in LIST:
    df[i] = df[i] * DATA

df = df.round(2)

# Save data
df.to_csv("data.csv", index=False)

# Preview the data
print(df.head())
